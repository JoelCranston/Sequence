printSeq(iterator,args)
    set numlength to 1
    if args.equalWidth is equal to True then
        set numlength tog length of longest possible number including sign.
        
    if args.format is equal to None then 
        for each x in iterator 
            print x with zeropadding of numlength
    else
        for each x in iterator
            print x with format string
